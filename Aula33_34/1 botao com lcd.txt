#include <LiquidCrystal_I2C.h>
//adiciona uma biblioteca
#define led 8
#define botao 12

LiquidCrystal_I2C lcd(0x27, 20, 4);//cria um objeto nome (enderecolcd, )

void setup() 
{
Serial.begin(9600);
pinMode(botao, INPUT);
pinMode(led, OUTPUT);
lcd.init();//inicia a biblioteca
lcd.backlight();//liga a luz de fundo do lcd
lcd.setCursor(0,0);//ecolhe onde vai ser escrito
lcd.print("estado led");
}

void loop()
{
static bool estadodobotaopassado;
bool estadodobotao = digitalRead(botao);
static bool estadodoled;
  
if(estadodobotao & !estadodobotaopassado)
{ estadodoled = !estadodoled;
  digitalWrite(led, estadodoled);
 lcd.clear();
 lcd.setCursor(0,0);
lcd.print("estado led");
 lcd.setCursor(0,1);
 lcd.print("on");}
  else if(estadodoled == 0)
  {lcd.setCursor(0,1);
 lcd.print("off");} 
estadodobotaopassado = estadodobotao;
}
