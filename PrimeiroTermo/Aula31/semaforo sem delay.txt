#define ledP 12
#define ledA 8
#define ledS 7

unsigned int tempms;
unsigned int tempinicial;

{
 pinMode(ledP, OUTPUT);
 pinMode(ledA, OUTPUT);
 pinMode(ledS, OUTPUT);
}

void loop()
{
 tempms = millis();
  if(3000 >= tempms-tempinicial)
 {digitalWrite(ledS,HIGH);
 digitalWrite(ledA,LOW);
 digitalWrite(ledP,LOW);}
 else if(5000 >= tempms-tempinicial)
 {digitalWrite(ledS,LOW);
 digitalWrite(ledA,HIGH);
 digitalWrite(ledP,LOW);}
  else if(10000 >= tempms-tempinicial)
 {digitalWrite(ledS,LOW);
 digitalWrite(ledA,LOW);
 digitalWrite(ledP,HIGH);}
  else
  {tempinicial = tempms;}
}
